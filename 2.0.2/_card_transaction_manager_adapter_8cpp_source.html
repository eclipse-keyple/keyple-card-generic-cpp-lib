102const std::vector<std::vector<uint8_t>> CardTransactionManagerAdapter::processApdusToByteArrays()
103{
104 std::shared_ptr<CardResponseApi> cardResponse;
105 std::vector<std::vector<uint8_t>> apduResponsesBytes;
106
107 if (mApduRequests.empty()) {
108 return apduResponsesBytes;
109 }
110
111 try {
112 cardResponse =
113 std::dynamic_pointer_cast<ProxyReaderApi>(mReader)
114 ->transmitCardRequest(
115 std::make_shared<CardRequestAdapter>(mApduRequests, false), mChannelControl);
116
117 } catch (const ReaderBrokenCommunicationException& e) {
118 throw TransactionException("Reader communication error.",
119 std::make_shared<ReaderBrokenCommunicationException>(e));
120 } catch (const CardBrokenCommunicationException& e) {
121 throw TransactionException("Card communication error.",
122 std::make_shared<CardBrokenCommunicationException>(e));
123 } catch (const UnexpectedStatusWordException& e) {
124 throw TransactionException("Apdu error.",
125 std::make_shared<UnexpectedStatusWordException>(e));
126 }
127
128 mApduRequests.clear();
129
130 for (const auto& apduResponse : cardResponse->getApduResponses()) {
131 apduResponsesBytes.push_back(apduResponse->getApdu());
132 }
133
134 return apduResponsesBytes;
135}

137const std::vector<std::string> CardTransactionManagerAdapter::processApdusToHexStrings()
138{
139 const std::vector<std::vector<uint8_t>> apduResponsesBytes = processApdusToByteArrays();
140 std::vector<std::string> apduResponsesHex;
141
142 for (const auto& bytes : apduResponsesBytes) {
143 apduResponsesHex.push_back(ByteArrayUtil::toHex(bytes));
144 }

146 return apduResponsesHex;
