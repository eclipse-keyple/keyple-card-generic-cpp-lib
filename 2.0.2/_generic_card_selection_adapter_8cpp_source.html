63GenericCardSelection& GenericCardSelectionAdapter::filterByPowerOnData(
64 const std::string& powerOnDataRegex)
65{
66 Assert::getInstance().notEmpty(powerOnDataRegex, "powerOnDataRegex");
67
68 try {
